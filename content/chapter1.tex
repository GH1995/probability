\subsection{条件概率}
\label{sub:条件概率}


在事件 $A$ 发生的情况下 $B$ 发生的概率
\[
    P(B|A) = \frac{P(AB)}{P(A)}
\]

\subsection{乘法公式}
\label{sub:乘法公式}

\[
    \begin{aligned}
        P(AB) = P(A) P(B\|A) \\
        P(A_1 A_2 \cdots A_n) = P(A_1)\, P(A_2 \| A_1) \, P(A_3 \| A_1 A_2) \cdots P(A_n \| A_1 \cdots A_{n-1})
    \end{aligned}
\]
